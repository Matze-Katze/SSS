def expoLine(x,a,b):
    return np.exp(a*np.log(x)+b)

def expoLineRange(a,b,liste):
    value = []
    for x in liste:
        value.append(expoLine(x,a,b))
    return value

plt.rcParams["figure.figsize"] = [5, 3.50]
